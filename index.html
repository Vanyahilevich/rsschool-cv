The Hashtag Generator

function generateHashtag (str) {
let re = /\s+/
let hashtags
if (str.length === 0) return false
let notSpace = str.split(re)
if(notSpace.join('') ===  '') return false
hashtags = notSpace.map(item => (item[0].toUpperCase() + item.slice(1)) ).join('')
if (hashtags.length >= 140) return false
if (hashtags === "") return ''
return `#${hashtags}`
